class TripletRiskService {
  async analyzeRiskForDraw(drawId) {
    // 1. Obtener el sorteo
    const draw = await prisma.draw.findUnique({
      where: { id: drawId },
      include: { game: true }
    });

    // 2. Obtener tripletas activas del juego
    const tripletas = await prisma.tripleBet.findMany({
      where: {
        gameId: draw.gameId,
        status: 'ACTIVE',
        createdAt: { lte: draw.drawDate }
      }
    });

    // 3. Obtener sorteos pasados del juego (desde creaciÃ³n de cada tripleta)
    const pastDraws = await prisma.draw.findMany({
      where: {
        gameId: draw.gameId,
        status: 'DRAWN',
        drawDate: { gte: /* fecha mÃ¡s antigua de tripletas */ }
      },
      select: { id: true, winnerItemId: true, drawDate: true }
    });

    // 4. Para cada tripleta, verificar cuÃ¡ntos nÃºmeros salieron
    const riskAnalysis = [];

    for (const tripleta of tripletas) {
      const numbers = [tripleta.item1Id, tripleta.item2Id, tripleta.item3Id];
      const matchedNumbers = [];

      // Revisar sorteos elegibles para esta tripleta
      const eligibleDraws = pastDraws.filter(d =>
        d.drawDate >= tripleta.createdAt
      ).slice(0, tripleta.totalDraws);

      // Contar cuÃ¡ntos de los 3 nÃºmeros ya salieron
      for (const pastDraw of eligibleDraws) {
        if (numbers.includes(pastDraw.winnerItemId)) {
          matchedNumbers.push(pastDraw.winnerItemId);
        }
      }

      // Si 2 de 3 ya salieron, hay riesgo
      if (matchedNumbers.length === 2) {
        const remainingNumber = numbers.find(n => !matchedNumbers.includes(n));

        riskAnalysis.push({
          tripletaId: tripleta.id,
          numbersMatched: matchedNumbers,
          remainingNumber: remainingNumber,
          exposure: tripleta.amount * tripleta.multiplier,
          userId: tripleta.userId
        });
      }
    }

    // 5. Agrupar por nÃºmero faltante
    const riskByNumber = {};
    for (const risk of riskAnalysis) {
      if (!riskByNumber[risk.remainingNumber]) {
        riskByNumber[risk.remainingNumber] = {
          itemId: risk.remainingNumber,
          totalExposure: 0,
          tripletasAtRisk: []
        };
      }
      riskByNumber[risk.remainingNumber].totalExposure += risk.exposure;
      riskByNumber[risk.remainingNumber].tripletasAtRisk.push(risk.tripletaId);
    }

    return {
      hasRisk: Object.keys(riskByNumber).length > 0,
      riskByNumber: riskByNumber,
      totalExposure: Object.values(riskByNumber).reduce((sum, r) => sum + r.totalExposure, 0)
    };
  }
}

// backend/src/middlewares/auth.middleware.js

const jwt = require('jsonwebtoken');

const authenticateToken = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: 'Token requerido' });
  }

  jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
    if (err) {
      return res.status(403).json({ error: 'Token invÃ¡lido' });
    }
    req.user = user;
    next();
  });
};

const requireRole = (roles) => {
  return (req, res, next) => {
    if (!roles.includes(req.user.role)) {
      return res.status(403).json({ error: 'Permisos insuficientes' });
    }
    next();
  };
};

module.exports = { authenticateToken, requireRole };
